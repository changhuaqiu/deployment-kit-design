### 场景1: 开发和迭代（P0）

#### 场景描述

开发者频繁修改 XaC 代码，需要快速验证和部署到测试环境。

#### 当前支持情况

**已支持**:
- ✅ discover-resources → generate-xac → update-xac → validate-syntax → test-deploy

**流程**:
```
1. 修改 XaC 代码
2. 执行语法校验
3. 部署到测试环境
4. 验证结果
```

#### 发现的问题

**问题1**: 缺少增量更新机制

**现状**: 每次都全量生成 XaC 代码，效率低

**影响**:
- 对于大型项目，生成耗时较长
- 开发迭代效率低

**解决方案**:

```python
class UpdateXacSkill(SkillBase):
    def execute(self, context):
        appid = context['appid']
        changes = context.get('changes')

        # 1. 检测变更
        if not changes:
            changes = self.detect_changes(appid)

        # 2. 增量更新
        if changes['type'] == 'incremental':
            # 只更新变更的资源
            updated_resources = self.update_resources(
                appid,
                changes['changed_resources']
            )
        else:
            # 全量更新
            updated_resources = self.regenerate_xac(appid)

        return {
            'type': changes['type'],
            'updated_resources': updated_resources
        }

    def detect_changes(self, appid):
        """检测变更"""
        # 对比缓存和现网资源
        cached = self.cache_manager.load(appid)
        current = self.call_mcp_service(appid)

        changed = []
        for resource_type in cached['resources']:
            if cached[resource_type] != current[resource_type]:
                changed.append(resource_type)

        if len(changed) < 5:
            return {'type': 'incremental', 'changed_resources': changed}
        else:
            return {'type': 'full', 'changed_resources': []}
```

class TestDeploySkill(SkillBase):

    def verify_deployment(self, appid):

            'resource_status': resource_status,
            'health_check': health_check,
            'test_result': test_result

#### 发现的问题

**问题4**: 环境管理能力分散，缺乏统一入口

**现状**: 环境相关功能分散在多个技能中

**影响**: 用户难以理解和使用

**解决方案**:

**统一的环境管理 skill**:

```python
class ManageEnvironmentSkill(SkillBase):
    """环境管理技能（聚合能力）"""

    def execute(self, context):
        action = context['action']
        appid = context['appid']

        if action == 'list':
            return self.list_environments(appid)
        elif action == 'status':
            return self.get_environment_status(appid, context['env'])
        elif action == 'compare':
            return self.compare_environments(appid, context['envs'])
        elif action == 'upgrade':
            return self.upgrade_environment(appid, context['source'], context['target'])
        else:
            raise ValueError(f"未知的环境操作: {action}")

    def list_environments(self, appid):
        """列出所有环境"""
        envs = self.call_mcp_service('get_environments', appid)
        return {
            'environments': envs,
            'total': len(envs)
        }

    def get_environment_status(self, appid, env):
        """获取环境状态"""
        status = self.call_mcp_service('get_environment_status', appid, env)
        return {
            'environment': env,
            'status': status['status'],
            'resources': status['resources'],
            'health': status['health']
        }

    def compare_environments(self, appid, envs):
        """对比环境配置"""
        # 获取两个环境的配置
        config1 = self.call_mcp_service('get_environment_config', appid, envs[0])
        config2 = self.call_mcp_service('get_environment_config', appid, envs[1])

        # 对比差异
        diff = self.diff_configs(config1, config2)

        return {
            'environments': envs,
            'differences': diff
        }

    def upgrade_environment(self, appid, source, target):
        """环境升级"""
        # 1. 对比源环境和目标环境
        comparison = self.compare_environments(appid, [source, target])

        # 2. 生成升级计划
        plan = self.generate_upgrade_plan(comparison)

        # 3. 执行升级
        result = self.execute_upgrade(plan)

        return result
```

### 场景3: 配置管理（P0）

#### 场景描述

#### 用户需求确认

- ✅ Git 集成（版本管理）
- ✅ 配置对比（diff）
- ✅ 配置合并（merge）

#### 发现的问题

**问题7**: 配置管理能力缺失

**现状**: 当前没有配置管理相关的技能

**影响**: 用户无法管理配置变更

**解决方案**:

**新增配置管理技能**:

```python
class ManageConfigSkill(SkillBase):
    """配置管理技能"""

    def execute(self, context):
        action = context['action']
        appid = context['appid']

        if action == 'init':
            return self.init_git_repo(appid)
        elif action == 'commit':
            return self.commit_config(appid, context['message'])
        elif action == 'diff':
            return self.diff_config(appid, context.get('ref'))
        elif action == 'merge':
            return self.merge_config(appid, context['source'], context['target'])
        elif action == 'rollback':
            return self.rollback_config(appid, context['ref'])
        else:
            raise ValueError(f"未知的配置操作: {action}")

    def init_git_repo(self, appid):
        """初始化 Git 仓库"""
        config_dir = self.get_config_dir(appid)

        # 初始化 Git 仓库
        repo = git.Repo.init(config_dir)

        # 创建 .gitignore
        gitignore = config_dir / '.gitignore'
        gitignore.write_text('.deployment-kit/cache/')

        # 首次提交
        repo.index.add(['xac_code/'])
        repo.index.commit('Initial commit')

        return {'status': 'initialized', 'repo': str(config_dir)}

    def commit_config(self, appid, message):
        """提交配置变更"""
        config_dir = self.get_config_dir(appid)
        repo = git.Repo(config_dir)

        # 检查变更
        if not repo.is_dirty():
            return {'status': 'no_changes'}

        # 添加所有变更
        repo.index.add(['xac_code/'])

        # 提交
        commit = repo.index.commit(message)

        return {
            'status': 'committed',
            'commit_hash': commit.hexsha,
            'message': message
        }

    def diff_config(self, appid, ref=None):
        """对比配置"""
        config_dir = self.get_config_dir(appid)
        repo = git.Repo(config_dir)

        # 获取差异
        if ref:
            # 对比指定版本
            diff = repo.git.diff(ref, 'HEAD')
        else:
            # 对比工作区
            diff = repo.git.diff()

        return {
            'diff': diff,
            'has_changes': len(diff) > 0
        }

    def merge_config(self, appid, source, target):
        """合并配置"""
        config_dir = self.get_config_dir(appid)
        repo = git.Repo(config_dir)

        # 切换到目标分支
        repo.git.checkout(target)

        # 合并源分支
        try:
            repo.git.merge(source)
            return {'status': 'merged', 'conflicts': False}
        except git.exc.GitCommandError as e:
            # 合并冲突
            return {
                'status': 'conflicts',
                'conflicts': True,
                'message': str(e)
            }

    def rollback_config(self, appid, ref):
        """回滚配置"""
        config_dir = self.get_config_dir(appid)
        repo = git.Repo(config_dir)

        # 回滚到指定版本
        repo.git.reset('--hard', ref)

        return {
            'status': 'rolled_back',
            'ref': ref
        }
```

class DryRunRehearsalSkill(SkillBase):
    """生产演练技能（增强版）"""

    def execute(self, context):
        appid = context['appid']
        mode = context.get('mode', 'plan')  # plan, sandbox, full

        if mode == 'plan':
            # 模式1: 计划验证（当前实现）
            return self.verify_plan(appid)
        elif mode == 'sandbox':
            # 模式2: 沙箱环境（新增）

            # 模式3: 完整演练（新增）

        sandbox_env = self.create_sandbox(appid)

        try:
            # 2. 在沙箱环境执行部署
            result = self.deploy_to_sandbox(appid, sandbox_env)

            # 3. 验证部署结果
            verification = self.verify_deployment(appid, sandbox_env)

            return {
                'mode': 'sandbox',
                'result': result,
                'verification': verification
            }
        finally:
            # 4. 清理沙箱环境
            self.cleanup_sandbox(sandbox_env)

    def full_rehearsal(self, appid):

        test_env = self.get_test_environment(appid)

        # 2. 执行完整部署流程

        # 3. 全面验证
        verification = self.verify_deployment(appid, test_env)

        # 4. 生成演练报告
        report = self.generate_rehearsal_report(result, verification)

        return {
            'mode': 'full',
            'result': result,
            'verification': verification,

class DeployProductionSkill(SkillBase):
    """生产部署技能（增强版）"""

    def execute(self, context):
        appid = context['appid']
        safety_checks = context.get('safety_checks', True)

        # 1. 前置检查
        if safety_checks:
            check_result = self.pre_deployment_checks(appid)
            if not check_result['passed']:
                return {
                    'status': 'failed',
                    'reason': 'Pre-deployment checks failed',
                    'check_result': check_result
                }

        # 2. 创建快照（用于回滚）
        snapshot = self.create_snapshot(appid)

        try:
            # 3. 执行部署
            result = self.deploy_to_production(appid)

            # 4. 后置验证
            verification = self.post_deployment_verification(appid)

            if verification['passed']:
                return {
                    'status': 'success',
                    'result': result,
                    'verification': verification
                }
            else:
                # 验证失败，自动回滚
                self.rollback_to_snapshot(appid, snapshot)
                return {
                    'status': 'rolled_back',
                    'reason': 'Post-deployment verification failed',
                    'verification': verification
                }
        except Exception as e:
            # 部署失败，自动回滚
            self.rollback_to_snapshot(appid, snapshot)
            raise

    def pre_deployment_checks(self, appid):
        """部署前检查"""
        checks = []

        # 1. 检查测试环境是否通过
        test_env_check = self.check_test_environment(appid)
        checks.append({'name': 'test_environment', 'passed': test_env_check})

        # 2. 检查资源配额
        quota_check = self.check_resource_quota(appid)
        checks.append({'name': 'resource_quota', 'passed': quota_check})

        # 3. 检查依赖服务
        dependency_check = self.check_dependencies(appid)
        checks.append({'name': 'dependencies', 'passed': dependency_check})

        # 4. 检查配置合法性
        config_check = self.check_config(appid)
        checks.append({'name': 'config', 'passed': config_check})

        passed = all(check['passed'] for check in checks)

        return {'passed': passed, 'checks': checks}

class RollbackDeploymentSkill(SkillBase):
    """回滚技能（增强版）"""

    def execute(self, context):
        appid = context['appid']
        strategy = context.get('strategy', 'full')

        if strategy == 'full':
            # 策略1: 完整回滚
            return self.full_rollback(appid, context['to'])
        elif strategy == 'partial':
            # 策略2: 部分回滚（只回滚指定资源）
            return self.partial_rollback(appid, context['resources'], context['to'])
        elif strategy == 'forward':
            # 策略3: 前向回滚（修复后重新部署）
            return self.forward_rollback(appid, context['fix'])

    def partial_rollback(self, appid, resources, to_version):
        """部分回滚"""
        # 1. 只回滚指定的资源
        for resource in resources:
            self.rollback_resource(appid, resource, to_version)

        # 2. 验证回滚结果
        verification = self.verify_rollback(appid, resources)

        return {
            'strategy': 'partial',
            'resources': resources,
            'to_version': to_version,
            'verification': verification
        }
