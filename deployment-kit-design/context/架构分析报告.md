### P0 问题（5个）- 必须解决

#### 问题1: 入口层与编排器职责重叠

**问题描述**:
当前设计中，Entry Layer（入口层）和 Orchestrator（编排器）职责存在重叠：
- 入口层负责：初始化检查、智能检查、自动修复
- 编排器负责：技能管理、工作流执行、上下文管理、错误处理

这两者的边界不够清晰，导致：
1. 检查逻辑分散在两层
2. 初始化和错误的职责不明确
3. 代码维护困难

**影响**:
- 架构层次不清晰
- 代码重复
- 难以维护

**解决方案**:

简化架构，移除独立的 Entry Layer，将其功能整合到 CLI 工具和 Orchestrator 中：

```
简化前:
用户 → CLI → Entry Layer → Orchestrator → Skills → MCP

简化后:
用户 → CLI → Orchestrator → Skills → MCP
         ↘智能检查→
```

**职责重新分配**:

1. **CLI 工具负责**:
   - 命令行参数解析
   - 配置文件加载
   - 基础环境检查（Python 版本、依赖安装）
   - 调用 Orchestrator

2. **Orchestrator 负责**:
   - 智能检查（依赖 MCP 的检查）
   - 技能管理
   - 工作流执行
   - 上下文管理
   - 错误处理和自动修复

**代码示例**:

```python
# cli.py
import argparse
from orchestrator import Orchestrator

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument('--appid', required=True)

    check_python_version()
    check_dependencies()

    # 创建编排器
    orchestrator = Orchestrator(project_dir='.')

    # 执行工作流
    orchestrator.run_workflow(
        workflow_name=args.workflow,
        params={'appid': args.appid}

class Orchestrator:
    def __init__(self, project_dir):
        self.project_dir = Path(project_dir)
        self.data_dir = self.project_dir / '.deployment-kit'
        self.skill_manager = SkillManager()
        self.context_manager = ContextManager()

    def run_workflow(self, workflow_name, params):
        # 智能检查（在这里执行，不在单独的 Entry Layer）
        self.smart_check(params)

        cache_dir = self.data_dir / 'cache' / params['appid']
        if cache_dir.exists():
            manifest = load_json(cache_dir / 'manifest.json')
            if is_cache_expired(manifest):
                print("⚠️  缓存已过期，建议刷新")
```

#### 问题3: 依赖关系未充分利用HEAM

**问题描述**:

```json
      "type": "explicit",
      "description": "应用依赖ADS集群部署",
      "depends_on": [
        {
          "resource_type": "ads",
          "resource_id": "ads-001",
          "relationship": "application_runs_on"
        }
      ]
    }
  ]
}
```

**问题**:
1. HEAM 协议可能已经定义了资源间的依赖关系
2. 重复定义容易冲突
3. 人工维护成本高

**影响**:
- 依赖关系可能与 HEAM 规范不一致
- 维护成本高
- 容易出错

**解决方案**:

**混合模式**（推荐）:

1. **从 HEAM 协议提取隐式依赖**
2. **允许人工显式依赖覆盖**
3. **依赖合并和冲突检测**

```python
def resolve_dependencies(appid, cache_dir, explicit_deps):
    """解析依赖关系（混合模式）"""

    # 1. 从 XaC 代码提取隐式依赖
    xac_code_dir = cache_dir / 'xac_code'
    implicit_deps = extract_dependencies_from_heam(xac_code_dir)

    # 2. 加载显式依赖（人工指定）
    explicit_deps = load_json('.deployment-kit/dependencies.json')

    # 3. 合并依赖
    merged_deps = merge_dependencies(implicit_deps, explicit_deps)

    # 4. 冲突检测
    conflicts = detect_conflicts(implicit_deps, explicit_deps)
    if conflicts:
        print(f"⚠️  发现依赖冲突:")
        for conflict in conflicts:
            print(f"  - {conflict}")
        print(f"  显式依赖将覆盖隐式依赖")

    return merged_deps

def extract_dependencies_from_heam(xac_code_dir):
    """从 HEAM 协议提取隐式依赖"""
    dependencies = []

    # 遍历 XaC 代码文件
    for xac_file in xac_code_dir.glob('**/*.tf'):
        # 解析 HEAM 资源定义
        resources = parse_heam_resources(xac_file)

        # 提取依赖关系
        for resource in resources:
            for ref in resource.get('depends_on', []):
                dependencies.append({
                    'type': 'implicit',
                    'source': resource['id'],
                    'target': ref,
                    'relationship': ref.get('relationship', 'unknown')
                })

    return dependencies
```

#### 问题5: MCP调用缺少并发和重试

**问题描述**:
当前 MCP 调用是串行的，没有并发和重试机制：

```python
def call_mcp_service(self, appid):
    """调用MCP服务"""
    # 1. 获取资源统计
    stats = mcp_call("get_resource_stats", appid)

    # 2. 调用各个tools获取详细数据（串行）
    resources = {}
    for resource_type in stats['types']:
        tool_name = f"get_{resource_type}_list"
        resources[resource_type] = mcp_call(tool_name, appid)

    return resources
```

**影响**:
- 性能差（串行调用慢）
- 可靠性差（单次失败即失败）
- 用户体验差

**解决方案**:

**并发调用 + 重试机制**:

```python
import asyncio
from tenacity import retry, stop_after_attempt, wait_exponential

class MCPCaller:
    """MCP 调用器（带并发和重试）"""

    def __init__(self, max_concurrent=5, max_retries=3):
        self.max_concurrent = max_concurrent
        self.max_retries = max_retries

    @retry(
        stop=stop_after_attempt(3),
        wait=wait_exponential(multiplier=1, min=1, max=10)
    )
    async def call_mcp_tool(self, tool_name, params):
        """调用单个 MCP 工具（带重试）"""
        try:
            result = await mcp_client.call(tool_name, params)
            return result
        except Exception as e:
            print(f"❌ MCP 调用失败: {tool_name}, 错误: {e}")
            raise

    async def fetch_all_resources(self, appid):
        """并发获取所有资源"""
        # 1. 获取资源统计
        stats = await self.call_mcp_tool("get_resource_stats", {"appid": appid})

        # 2. 并发获取各类资源
        tasks = []
        for resource_type in stats['types']:
            tool_name = f"get_{resource_type}_list"
            task = self.call_mcp_tool(tool_name, {"appid": appid})
            tasks.append(task)

        # 3. 等待所有任务完成
        results = await asyncio.gather(*tasks, return_exceptions=True)

        # 4. 处理结果
        resources = {}
        for i, result in enumerate(results):
            resource_type = stats['types'][i]
            if isinstance(result, Exception):
                print(f"⚠️  获取 {resource_type} 失败: {result}")
                resources[resource_type] = None
            else:
                resources[resource_type] = result

        return resources

# 在 discover-resources skill 中使用
class DiscoverResourcesSkill(SkillBase):
    async def execute(self, context):
        appid = context['appid']

        # 创建 MCP 调用器
        mcp_caller = MCPCaller(max_concurrent=5, max_retries=3)

        # 并发获取资源
        resources = await mcp_caller.fetch_all_resources(appid)

        # 保存到缓存
        self.save_to_cache(appid, resources)

        return resources
```

```python
class WorkflowComposer:
    """工作流组合器"""

    def compose(self, skills, conditions=None):
        """动态组合工作流"""
        workflow = {
            'name': f'custom-{len(skills)}-steps',
            'skills': skills,
            'conditions': conditions or {},
            'gates': []
        }
        return workflow

    def add_gate(self, workflow, skill_name, gate_type, conditions):
        """添加门控"""
        workflow['gates'].append({
            'after': skill_name,
            'type': gate_type,
            'conditions': conditions
        })
        return workflow

    workflow,
    'validate-syntax',
    gate_type='manual_approval',
    conditions={'changes': 'large'}
```

#### 问题9: 错误处理机制不够健壮

**问题描述**:
当前错误处理逻辑简单，只是捕获异常并打印：

```python
try:
    result = skill.execute(context)
except Exception as e:
    print(f"技能执行失败: {e}")
    return None
```

**缺少**:
- 错误分类
- 自动修复尝试
- 降级策略
- 详细日志

**解决方案**:

**分类错误处理**：

```python
class ErrorHandler:
    """错误处理器"""

    def __init__(self):
        self.error_handlers = {
            'MCPConnectionError': self.handle_mcp_connection,
            'MCPTimeoutError': self.handle_mcp_timeout,
            'ValidationError': self.handle_validation,
            'CacheError': self.handle_cache,
            'SkillExecutionError': self.handle_skill_execution
        }

    def handle(self, error, context):
        """处理错误"""
        error_type = type(error).__name__

        # 1. 查找对应的处理器
        handler = self.error_handlers.get(error_type)
        if handler:
            return handler(error, context)
        else:
            # 默认处理
            return self.handle_unknown(error, context)

    def handle_mcp_timeout(self, error, context):
        """处理 MCP 超时"""
        print(f"⚠️  MCP 超时，尝试重试...")
        # 自动重试
        return {'action': 'retry', 'max_attempts': 3}

    def handle_validation(self, error, context):
        """处理验证错误"""
        print(f"❌ 验证失败: {error.message}")
        # 不自动修复，需要人工介入
        return {'action': 'manual', 'reason': 'validation_failed'}

    def handle_skill_execution(self, error, context):
        """处理技能执行错误"""
        print(f"❌ 技能执行失败: {error.message}")

        # 尝试自动修复
        if error.auto_fixable:
```

```python
from abc import ABC, abstractmethod
from typing import Dict, Any, Optional
from pathlib import Path

class SkillBase(ABC):
    """技能基类"""

    def __init__(self, metadata: Dict[str, Any]):
        self.name = metadata['name']
        self.version = metadata['version']
        self.description = metadata['description']
        self.author = metadata.get('author', 'unknown')
        self.pre_conditions = metadata.get('pre_conditions', {})
        self.capabilities = metadata.get('capabilities', [])

    @abstractmethod
    def execute(self, context: Dict[str, Any]) -> Dict[str, Any]:
        """执行技能（子类必须实现）"""
        pass

    def validate_pre_conditions(self, context: Dict[str, Any]) -> bool:
        """验证前置条件"""
        for condition_name, condition_value in self.pre_conditions.items():
            if condition_name == 'requires_cache':
                if not context.get('cache_exists'):
                    raise ValueError(f"技能 {self.name} 需要缓存，但缓存不存在")
            # 更多条件检查...
        return True

    def before_execute(self, context: Dict[str, Any]) -> None:
        """执行前钩子（可选重写）"""
        pass

    def after_execute(self, context: Dict[str, Any], result: Dict[str, Any]) -> None:
```

#### 2. 技能加载器（core/skill_loader.py）

```python
import importlib.util
import yaml
from pathlib import Path
from typing import Dict

class SkillLoader:
    """技能加载器"""

    def __init__(self, skills_dir: Path):
        self.skills_dir = Path(skills_dir)
        self.skills_cache = {}

    def load_all_skills(self) -> Dict[str, SkillBase]:
        """加载所有技能"""
        if self.skills_cache:
            return self.skills_cache

        skills = {}
        for skill_dir in self.skills_dir.iterdir():
            if skill_dir.is_dir() and (skill_dir / 'skill.yaml').exists():
                skill = self.load_skill(skill_dir)
                skills[skill.name] = skill

        self.skills_cache = skills
        return skills

    def load_skill(self, skill_dir: Path) -> SkillBase:
        """加载单个技能"""
        # 1. 读取技能元数据
        metadata_file = skill_dir / 'skill.yaml'
        with open(metadata_file, 'r', encoding='utf-8') as f:
            metadata = yaml.safe_load(f)

        # 2. 动态加载技能模块
        module_path = skill_dir / 'main.py'
        spec = importlib.util.spec_from_file_location(
            f"skill_{metadata['name']}",
            module_path
        )
        module = importlib.util.module_from_spec(spec)
        spec.loader.exec_module(module)

        # 3. 获取技能类并实例化
        skill_class = getattr(module, 'Skill')
        skill = skill_class(metadata)

        return skill

    def reload_skill(self, skill_name: str) -> SkillBase:
        """重新加载技能（用于开发调试）"""
        skill_dir = self.skills_dir / skill_name
        if skill_name in self.skills_cache:
            del self.skills_cache[skill_name]
        return self.load_skill(skill_dir)
```

#### 3. 编排器（core/orchestrator.py）

```python
from typing import Dict, Any, List
from pathlib import Path

class Orchestrator:
    """编排器（核心骨架）"""

    def __init__(self, project_dir: Path, skills_dir: Path):
        self.project_dir = Path(project_dir)
        self.data_dir = self.project_dir / '.deployment-kit'
        self.skill_loader = SkillLoader(skills_dir)
        self.cache_manager = CacheManager(self.data_dir / 'cache')
        self.context_manager = ContextManager(self.data_dir)

        # 加载所有技能（动态加载，不需要硬编码）
        self.skills = self.skill_loader.load_all_skills()

    def run_workflow(self, workflow: List[str], context: Dict[str, Any]) -> Dict[str, Any]:
        """执行工作流"""
        results = []

        for skill_name in workflow:
            if skill_name not in self.skills:
                raise ValueError(f"技能不存在: {skill_name}")

            skill = self.skills[skill_name]

            # 验证前置条件
            if not skill.validate_pre_conditions(context):
                raise ValueError(f"技能 {skill_name} 前置条件不满足")

            # 执行技能
            print(f"→ 执行技能: {skill_name}")
            result = self.execute_skill(skill, context)
            results.append(result)

            # 更新上下文
            context = self.context_manager.update(context, result)

        return {'results': results, 'status': 'success'}

    def execute_skill(self, skill: SkillBase, context: Dict[str, Any]) -> Dict[str, Any]:
        """执行单个技能"""
        try:
            # 执行前钩子
            skill.before_execute(context)

            # 执行技能
            result = skill.execute(context)

            # 执行后钩子
            skill.after_execute(context, result)

            return result
        except Exception as e:
            # 错误处理
            skill.on_error(context, e)
            raise
```
